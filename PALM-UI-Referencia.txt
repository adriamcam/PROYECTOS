@code {
    private PalmReportData? reportData;
    private bool isLoading;
    private string? errorMessage;
    private DateTime? lastLoadedAt;

    protected override async Task OnInitializedAsync()
    {
        await LoadReportAsync();
    }

    private async Task LoadReportAsync()
    {
        if (isLoading)
        {
            return;
        }

        try
        {
            isLoading = true;
            errorMessage = null;

            reportData = await PalmService.GetReportAsync();
            lastLoadedAt = DateTime.Now;
        }
        catch (Exception ex)
        {
            errorMessage =
                "Se produjo un error consultando las vistas PALM. " +
                ex.Message;
        }
        finally
        {
            isLoading = false;
        }
    }
}

    [Parameter]
    public PalmDashboard Dashboard { get; set; } = new();

    private static string FormatNumber(long value)
    {
        return value.ToString("N0");
    }

    private static string FormatNullableDateTime(DateTime? value)
    {
        return value.HasValue
            ? value.Value.ToString("dd/MM/yyyy HH:mm")
            : "No disponible";
    }

    private int GetComplianceAriaValue()
    {
        return Convert.ToInt32(
            Math.Clamp(Dashboard.CompliancePercent, 0m, 100m));
    }

    private string GetComplianceStyle()
    {
        var percentage = Math.Clamp(
            Dashboard.CompliancePercent,
            0m,
            100m);

        return FormattableString.Invariant(
            $"width: {percentage}%;");
    }
}

@code {
    [Parameter]
    public PalmRun? LatestRun { get; set; }

    private static string FormatDateTime(DateTime value)
    {
        return value.ToString("dd/MM/yyyy HH:mm:ss");
    }

    private static string FormatDateTime(DateTime? value)
    {
        return value.HasValue
            ? value.Value.ToString("dd/MM/yyyy HH:mm:ss")
            : "En proceso";
    }

    private static string FormatDuration(int? seconds)
    {
        if (!seconds.HasValue)
        {
            return "No disponible";
        }

        var duration = TimeSpan.FromSeconds(seconds.Value);

        if (duration.TotalHours >= 1)
        {
            return $"{(int)duration.TotalHours} h " +
                   $"{duration.Minutes} min " +
                   $"{duration.Seconds} s";
        }

        if (duration.TotalMinutes >= 1)
        {
            return $"{duration.Minutes} min " +
                   $"{duration.Seconds} s";
        }

        return $"{duration.Seconds} s";
    }

    private static string GetStatusText(string? status)
    {
        return status?.Trim().ToUpperInvariant() switch
        {
            "COMPLETED" => "Completado",
            "COMPLETEDWITHERRORS" => "Completado con errores",
            "FAILED" => "Fallido",
            "RUNNING" => "En ejecución",
            _ => string.IsNullOrWhiteSpace(status)
                ? "Sin estado"
                : status
        };
    }

    private static string GetStatusClass(string? status)
    {
        return status?.Trim().ToUpperInvariant() switch
        {
            "COMPLETED" =>
                "palm-information-status status-success",

            "COMPLETEDWITHERRORS" =>
                "palm-information-status status-warning",

            "FAILED" =>
                "palm-information-status status-danger",

            "RUNNING" =>
                "palm-information-status status-info",

            _ =>
                "palm-information-status status-neutral"
        };
    }
}

    [Parameter]
    public IReadOnlyList<PalmResource> Resources { get; set; }
        = Array.Empty<PalmResource>();

    private string searchText = string.Empty;
    private string selectedStatus = string.Empty;

    private IEnumerable<PalmResource> FilteredResources =>
        Resources
            .Where(MatchesStatus)
            .Where(MatchesSearch)
            .OrderBy(resource => resource.CustomerName)
            .ThenBy(resource => resource.SubscriptionName);

    private bool MatchesStatus(PalmResource resource)
    {
        return string.IsNullOrWhiteSpace(selectedStatus) ||
               string.Equals(
                   resource.Status,
                   selectedStatus,
                   StringComparison.OrdinalIgnoreCase);
    }

    private bool MatchesSearch(PalmResource resource)
    {
        if (string.IsNullOrWhiteSpace(searchText))
        {
            return true;
        }

        var value = searchText.Trim();

        return Contains(resource.CustomerName, value) ||
               Contains(resource.TenantId.ToString(), value) ||
               Contains(resource.SubscriptionName, value) ||
               Contains(resource.SubscriptionId?.ToString(), value) ||
               Contains(resource.PartnerIdBefore, value) ||
               Contains(resource.PartnerIdCurrent, value) ||
               Contains(resource.PartnerIdTarget, value) ||
               Contains(resource.PartnerValidationStatus, value) ||
               Contains(resource.Action, value) ||
               Contains(resource.ErrorMessage, value);
    }

    private static bool Contains(string? source, string value)
    {
        return !string.IsNullOrWhiteSpace(source) &&
               source.Contains(
                   value,
                   StringComparison.OrdinalIgnoreCase);
    }

    private static string FormatText(string? value)
    {
        return string.IsNullOrWhiteSpace(value)
            ? "No disponible"
            : value;
    }

    private static string FormatGuid(Guid? value)
    {
        return value.HasValue && value.Value != Guid.Empty
            ? value.Value.ToString()
            : "No disponible";
    }

    private static string GetRowClass(PalmResource resource)
    {
        return resource.RequiresAction
            ? "palm-row-requires-action"
            : string.Empty;
    }

    private static string GetStatusBadge(string? status)
    {
        return status?.Trim().ToUpperInvariant() switch
        {
            "OK" =>
                "badge text-bg-success",

            "NOK" =>
                "badge text-bg-danger",

            _ =>
                "badge text-bg-secondary"
        };
    }

    private static string GetValidationBadge(string? status)
    {
        var normalized = status?.Trim().ToUpperInvariant();

        return normalized switch
        {
            "CORRECT" or "OK" or "VALID" =>
                "badge text-bg-success",

            "DIFFERENT" or "MISMATCH" =>
                "badge text-bg-warning",

            "MISSING" or "WITHOUTPARTNERID" or "NOK" =>
                "badge text-bg-danger",

            _ =>
                "badge text-bg-secondary"
        };
    }

    [Parameter]

        = Array.Empty<PalmRun>();

    private IEnumerable<PalmRun> OrderedRuns =>
        Runs.OrderByDescending(run => run.StartedAt);

    private static string FormatDateTime(DateTime? value)
    {
        return value.HasValue
            ? value.Value.ToString("dd/MM/yyyy HH:mm:ss")
            : "En proceso";
    }

    private static string FormatDuration(int? seconds)
    {
        if (!seconds.HasValue)
        {
            return "No disponible";
        }

        var duration = TimeSpan.FromSeconds(seconds.Value);

        if (duration.TotalHours >= 1)
        {
            return $"{(int)duration.TotalHours} h " +
                   $"{duration.Minutes} min " +
                   $"{duration.Seconds} s";
        }

        if (duration.TotalMinutes >= 1)
        {
            return $"{duration.Minutes} min " +
                   $"{duration.Seconds} s";
        }

        return $"{duration.Seconds} s";
    }

    private static string FormatRunId(Guid runId)
    {
        var value = runId.ToString();

        return value.Length > 13
            ? $"{value[..8]}...{value[^4..]}"
            : value;
    }

    private bool ShowInformation { get; set; }

    private void OpenInformation()
    {
        ShowInformation = true;
    }

    private void CloseInformation()
    {
        ShowInformation = false;
    }

private ReservedInstanceDashboard Dashboard { get; set; } = new();
    private RICoverageKpi CoverageKpis { get; set; } = new();
    private List<RICoverageResource> CoverageResources { get; set; } = new();
    private bool IsLoading { get; set; } = true;
    private bool IsLoaded { get; set; }
    private string ActiveTab { get; set; } = "Dashboard";
    private string ResourcesInitialFilter { get; set; } = "";

    protected override async Task OnInitializedAsync()
    {
        await LoadAsync();
    }

    private void OpenResources(string filter)
    {
        ResourcesInitialFilter = filter;
        ActiveTab = "Resources";
    }

    private void OpenAction()
    {
        ActiveTab = "Action";
    }

    private void ChangeTab(string tab)
    {
        ActiveTab = tab;
    }

    private async Task LoadAsync()
    {
        IsLoading = true;

        try
        {
            var dashboardTask =
                ReservedInstanceService.GetDashboardAsync();

            var coverageKpisTask =
                ReservedInstanceService.GetCoverageKpisAsync();

            var coverageResourcesTask =
                ReservedInstanceService.GetCoverageResourcesAsync();

            await Task.WhenAll(
                dashboardTask,
                coverageKpisTask,
                coverageResourcesTask
            );

            Dashboard = await dashboardTask;
            CoverageKpis = await coverageKpisTask;
            CoverageResources = await coverageResourcesTask;

            IsLoaded = true;
        }
        finally
        {
            IsLoading = false;
        }
    }

    private static string GetDistributionClass(string status) => status switch
    {
        "Healthy" => "success",
        "SKU cambiado" => "warning",
        "Fecha no coincide" => "danger",
        "Baja utilización" => "purple",
        "Por vencer" => "info",
        "Expiradas" => "muted",
        _ => "primary"
    };
}

    [Parameter] public List<ReservedInstanceResource> Resources { get; set; } = new();
    [Parameter] public string InitialFilter { get; set; } = "";
    [Parameter] public EventCallback<string> OnChangeTab { get; set; }

    private string SubscriptionFilter { get; set; } = "";
    private string StatusFilter { get; set; } = "";
    private string QuickFilter { get; set; } = "";
    private string SearchText { get; set; } = "";

    private ReservedInstanceResource? SelectedResource { get; set; }

    private int CurrentPage { get; set; } = 1;
    private int PageSize { get; set; } = 20;
    private string SortColumn { get; set; } = "Status";
    private bool SortAscending { get; set; } = true;

    private IEnumerable<string> SubscriptionOptions =>
        Resources.Select(x => x.Subscription)
            .Where(x => !string.IsNullOrWhiteSpace(x))
            .Distinct()
            .OrderBy(x => x);

    private IEnumerable<ReservedInstanceResource> QuickFilteredResources =>
        QuickFilter switch
        {
            "Action" => Resources.Where(x => x.OperationalStatus != "Healthy"),
            "Healthy" => Resources.Where(x => x.OperationalStatus == "Healthy"),
            "ExpiringSoon" => Resources.Where(x => x.ExpiresSoon == true || x.OperationalStatus == "Expiring Soon"),
            _ => Resources
        };

    private IEnumerable<ReservedInstanceResource> FilteredResources =>
        QuickFilteredResources.Where(x =>
            (string.IsNullOrWhiteSpace(SubscriptionFilter) || x.Subscription == SubscriptionFilter) &&
            (string.IsNullOrWhiteSpace(StatusFilter) || x.OperationalStatus == StatusFilter) &&
            (
                string.IsNullOrWhiteSpace(SearchText) ||
                ContainsSafe(x.ResourceName, SearchText) ||
                ContainsSafe(x.Customer, SearchText) ||
                ContainsSafe(x.Subscription, SearchText) ||
                ContainsSafe(x.SKUReserva, SearchText) ||
                ContainsSafe(x.SKUVM, SearchText)
            )
        );

    private IEnumerable<ReservedInstanceResource> OrderedResources =>
        SortColumn switch
        {
            "Subscription" => SortAscending ? FilteredResources.OrderBy(x => x.Subscription) : FilteredResources.OrderByDescending(x => x.Subscription),
            "Resource" => SortAscending ? FilteredResources.OrderBy(x => x.ResourceName) : FilteredResources.OrderByDescending(x => x.ResourceName),
            "SKUReserva" => SortAscending ? FilteredResources.OrderBy(x => x.SKUReserva) : FilteredResources.OrderByDescending(x => x.SKUReserva),
            "SKUVM" => SortAscending ? FilteredResources.OrderBy(x => x.SKUVM) : FilteredResources.OrderByDescending(x => x.SKUVM),
            "FechaReserva" => SortAscending ? FilteredResources.OrderBy(x => x.FechaReserva) : FilteredResources.OrderByDescending(x => x.FechaReserva),
            "FechaVM" => SortAscending ? FilteredResources.OrderBy(x => x.FechaVM) : FilteredResources.OrderByDescending(x => x.FechaVM),
            "Days" => SortAscending ? FilteredResources.OrderBy(x => x.DaysRemaining) : FilteredResources.OrderByDescending(x => x.DaysRemaining),
            "Utilization" => SortAscending ? FilteredResources.OrderBy(x => x.Utilization30d) : FilteredResources.OrderByDescending(x => x.Utilization30d),
            "Status" => SortAscending ? FilteredResources.OrderBy(x => x.OperationalStatus) : FilteredResources.OrderByDescending(x => x.OperationalStatus),
            _ => FilteredResources.OrderBy(x => x.Subscription).ThenBy(x => x.ResourceName)
        };
    private IEnumerable<ReservedInstanceResource> PagedResources =>
        OrderedResources.Skip((CurrentPage - 1) * PageSize).Take(PageSize);

    private int TotalRecords => FilteredResources.Count();

    private int TotalPages =>
        Math.Max(1, (int)Math.Ceiling(TotalRecords / (double)PageSize));

    private IEnumerable<int> VisiblePages
    {
        get
        {
            var start = Math.Max(1, CurrentPage - 2);
            var end = Math.Min(TotalPages, CurrentPage + 2);

            for (var i = start; i <= end; i++)
            {
                yield return i;
            }
        }
    }

    private void SortBy(string column)
    {
        if (SortColumn == column)
        {
            SortAscending = !SortAscending;
        }
        else
        {
            SortColumn = column;
            SortAscending = true;
        }

        CurrentPage = 1;
    }

    private string GetSortIcon(string column)
    {
        if (SortColumn != column)
        {
            return "↕";
        }

        return SortAscending ? "↑" : "↓";
    }

    private void GoToPage(int pageNumber) => CurrentPage = pageNumber;
    private void FirstPage() => CurrentPage = 1;
    private void LastPage() => CurrentPage = TotalPages;
    private void NextPage() { if (CurrentPage < TotalPages) CurrentPage++; }
    private void PrevPage() { if (CurrentPage > 1) CurrentPage--; }

    private async Task OpenResourcesTab()
    {
        await OnChangeTab.InvokeAsync("Resources");
    }

    private async Task OpenActionTab()
    {
        await OnChangeTab.InvokeAsync("Action");
    }

    protected override void OnParametersSet()
    {
        if (!string.IsNullOrWhiteSpace(InitialFilter))
        {
            QuickFilter = InitialFilter;
            StatusFilter = "";
            CurrentPage = 1;
        }
    }

    private void ApplyQuickFilter(string filter)
    {
        QuickFilter = filter;
        StatusFilter = "";
        SubscriptionFilter = "";
        SearchText = "";
        CurrentPage = 1;
    }

    private void ClearFilters()
    {
        SubscriptionFilter = "";
        StatusFilter = "";
        QuickFilter = "";
        SearchText = "";
        QuickFilter = "";
        CurrentPage = 1;
    }

    private void ChangePageSize(ChangeEventArgs e)
    {
        if (int.TryParse(e.Value?.ToString(), out var value))
        {
            PageSize = value;
            CurrentPage = 1;
        }
    }

    private void OpenDrawer(ReservedInstanceResource item) => SelectedResource = item;
    private void CloseDrawer() => SelectedResource = null;

    private string GetActiveFiltersText()
    {
        var filters = new List<string>();

        if (!string.IsNullOrWhiteSpace(SubscriptionFilter)) filters.Add($"Suscripción: {SubscriptionFilter}");
        if (!string.IsNullOrWhiteSpace(StatusFilter)) filters.Add($"Estado: {GetStatusText(StatusFilter)}");
        if (!string.IsNullOrWhiteSpace(SearchText)) filters.Add($"Búsqueda: {SearchText}");

        return filters.Any() ? string.Join(" | ", filters) : "Ninguno";
    }

    private async Task ExportCsvAsync()
    {
        var sb = new StringBuilder();
        sb.AppendLine("Estado,Recurso,Cliente,Suscripcion,SKUReserva,SKUActual,FechaReserva,FechaVM,Dias,Utilizacion,Recomendacion,CambiadoPor,FechaCambio");

        foreach (var item in OrderedResources)
        {
            sb.AppendLine(string.Join(",",
                Csv(GetStatusText(item.OperationalStatus)),
                Csv(item.ResourceName),
                Csv(item.Customer),
                Csv(item.Subscription),
                Csv(item.SKUReserva),
                Csv(item.SKUVM),
                Csv(item.FechaReserva?.ToString("dd/MM/yyyy") ?? "-"),
                Csv(item.FechaVM?.ToString("dd/MM/yyyy") ?? "-"),
                Csv(item.DaysRemaining?.ToString() ?? "-"),
                Csv((item.Utilization30d ?? 0).ToString("N2")),
                Csv(GetRecommendation(item)),
                Csv(string.IsNullOrWhiteSpace(item.LastChangedBy) ? "Sin dato" : item.LastChangedBy),
                Csv(item.LastChangeDate?.ToString("dd/MM/yyyy HH:mm") ?? "-")
            ));
        }

        var fileName = $"reserved-instances-{DateTime.Now:yyyyMMddHHmm}.csv";
        var data = Uri.EscapeDataString(sb.ToString());

        await JS.InvokeVoidAsync("eval", $"const a=document.createElement('a');a.href='data:text/csv;charset=utf-8,{data}';a.download='{fileName}';document.body.appendChild(a);a.click();a.remove();");
    }

    private static string Csv(string? value)

    private static bool ContainsSafe(string? value, string search) =>
        !string.IsNullOrWhiteSpace(value) &&

@code {
    [Parameter] public List<ReservedInstanceResource> Resources { get; set; } = new();

    private ReservedInstanceResource? SelectedResource { get; set; }
    private List<ReservedInstanceChangeHistory> ChangeHistory { get; set; } = new();

    private string SearchText { get; set; } = "";
    
    private ReservedInstanceResource? SelectedNoteResource { get; set; }
    private string NoteText { get; set; } = "";
private string ProblemFilter { get; set; } = "";
    private string PriorityFilter { get; set; } = "";
    private string SubscriptionFilter { get; set; } = "";

    private int CurrentPage { get; set; } = 1;
    private int PageSize { get; set; } = 20;
    private string SortColumn { get; set; } = "Priority";
    private bool SortAscending { get; set; } = true;

    private List<ReservedInstanceResource> ActionResources =>
        Resources
            .Where(x => x.OperationalStatus != "Healthy")
            .OrderBy(GetPriorityRank)
            .ThenBy(x => x.Customer)
            .ThenBy(x => x.ResourceName)
            .ToList();

    private IEnumerable<string> SubscriptionOptions =>
        ActionResources
            .Select(x => x.Subscription)
            .Where(x => !string.IsNullOrWhiteSpace(x))
            .Distinct()
            .OrderBy(x => x);

    private IEnumerable<ReservedInstanceResource> FilteredResources =>
        ActionResources.Where(x =>
            (string.IsNullOrWhiteSpace(ProblemFilter) || x.OperationalStatus == ProblemFilter) &&
            (string.IsNullOrWhiteSpace(PriorityFilter) || GetPriority(x) == PriorityFilter) &&
            (string.IsNullOrWhiteSpace(SubscriptionFilter) || x.Subscription == SubscriptionFilter) &&
            (
                string.IsNullOrWhiteSpace(SearchText) ||
                ContainsSafe(x.ResourceName, SearchText) ||
                ContainsSafe(x.Customer, SearchText) ||
                ContainsSafe(x.Subscription, SearchText) ||
                ContainsSafe(x.SKUReserva, SearchText) ||
                ContainsSafe(x.SKUVM, SearchText)
            )
        );


    private IEnumerable<ReservedInstanceResource> OrderedResources =>
        SortColumn switch
        {
            "Priority" => SortAscending ? FilteredResources.OrderBy(GetPriorityRank) : FilteredResources.OrderByDescending(GetPriorityRank),
            "Subscription" => SortAscending ? FilteredResources.OrderBy(x => x.Subscription) : FilteredResources.OrderByDescending(x => x.Subscription),
            "Resource" => SortAscending ? FilteredResources.OrderBy(x => x.ResourceName) : FilteredResources.OrderByDescending(x => x.ResourceName),
            "SKUReserva" => SortAscending ? FilteredResources.OrderBy(x => x.SKUReserva) : FilteredResources.OrderByDescending(x => x.SKUReserva),
            "SKUVM" => SortAscending ? FilteredResources.OrderBy(x => x.SKUVM) : FilteredResources.OrderByDescending(x => x.SKUVM),
            "Problem" => SortAscending ? FilteredResources.OrderBy(GetProblem) : FilteredResources.OrderByDescending(GetProblem),
            "Detail" => SortAscending ? FilteredResources.OrderBy(GetDetail) : FilteredResources.OrderByDescending(GetDetail),
            "LifecycleRecommendation" => SortAscending
                ? FilteredResources.OrderBy(x => x.LifecycleRecommendation)
                : FilteredResources.OrderByDescending(x => x.LifecycleRecommendation),

            "ChangedBy" => SortAscending ? FilteredResources.OrderBy(x => x.LastChangedBy) : FilteredResources.OrderByDescending(x => x.LastChangedBy),
            "ChangeDate" => SortAscending ? FilteredResources.OrderBy(x => x.LastChangeDate) : FilteredResources.OrderByDescending(x => x.LastChangeDate),
            _ => FilteredResources.OrderBy(GetPriorityRank)
        };
    private IEnumerable<ReservedInstanceResource> PagedResources =>
        OrderedResources.Skip((CurrentPage - 1) * PageSize).Take(PageSize);

    private int TotalPages =>
        Math.Max(1, (int)Math.Ceiling(FilteredResources.Count() / (double)PageSize));

    private IEnumerable<int> VisiblePages
    {
        get
        {
            var start = Math.Max(1, CurrentPage - 2);
            var end = Math.Min(TotalPages, CurrentPage + 2);

            for (var i = start; i <= end; i++)
                yield return i;
        }
    }


    private void SortBy(string column)
    {
        if (SortColumn == column)
        {
            SortAscending = !SortAscending;
        }
        else
        {
            SortColumn = column;

        foreach (var item in FilteredResources)

                Csv(item.Customer),
                Csv(item.Subscription),
                Csv(item.ResourceName),
                Csv(item.SKUReserva),
                Csv(item.SKUVM),
                Csv(GetProblem(item)),
                Csv(GetDetail(item)),
                Csv(item.DaysRemaining?.ToString() ?? "-"),
                Csv(string.IsNullOrWhiteSpace(item.LastChangedBy) ? "Sin dato" : item.LastChangedBy),
                Csv(item.LastChangeDate?.ToString("dd/MM/yyyy HH:mm") ?? "-")
            ));
        }

        var fileName = $"ri-requiere-accion-{DateTime.Now:yyyyMMddHHmm}.csv";
        var data = Uri.EscapeDataString(sb.ToString());

        await JS.InvokeVoidAsync("eval", $"const a=document.createElement('a');a.href='data:text/csv;charset=utf-8,{data}';a.download='{fileName}';document.body.appendChild(a);a.click();a.remove();");
    }

    [Parameter]
    public List<HybridBenefitResource> Resources { get; set; } = new();

    private string SearchText { get; set; } = "";
    private string SubscriptionSearch { get; set; } = "";

    private IEnumerable<string> SubscriptionOptions =>
        Resources
            .Select(x => x.Subscription)
            .Where(x => !string.IsNullOrWhiteSpace(x))
            .Distinct()
            .OrderBy(x => x);
    private string StatusFilter { get; set; } = "";
    private string TypeFilter { get; set; } = "";
    private string SeverityFilter { get; set; } = "";

    private int CurrentPage { get; set; } = 1;
    private int PageSize { get; set; } = 25;

    private string SortColumn { get; set; } = "Resource";
    private bool SortAscending { get; set; } = true;

    private bool ShowHistoryPanel { get; set; }
    private bool IsLoadingHistory { get; set; }
    private HybridBenefitResource? SelectedResource { get; set; }
    private List<HybridBenefitResourceHistory> ResourceHistory { get; set; } = new();

    private IEnumerable<HybridBenefitResource> FilteredResources =>
        Resources.Where(x =>
            (string.IsNullOrWhiteSpace(SearchText)
                || x.Customer.Contains(SearchText, StringComparison.OrdinalIgnoreCase)
                || x.ResourceGroup.Contains(SearchText, StringComparison.OrdinalIgnoreCase)
                || x.ResourceName.Contains(SearchText, StringComparison.OrdinalIgnoreCase))
            &&
            (string.IsNullOrWhiteSpace(SubscriptionSearch)
                || x.Subscription.Contains(SubscriptionSearch, StringComparison.OrdinalIgnoreCase))
            &&
            (string.IsNullOrWhiteSpace(StatusFilter)
                || x.HealthStatus.Equals(StatusFilter, StringComparison.OrdinalIgnoreCase))
            &&
            (string.IsNullOrWhiteSpace(SeverityFilter)
                || x.Severity.Equals(SeverityFilter, StringComparison.OrdinalIgnoreCase))
            &&
            (
                string.IsNullOrWhiteSpace(TypeFilter)
                || (TypeFilter == "Windows" && x.HasWindowsAHUB && !x.HasSqlAHUB)
                || (TypeFilter == "SQL" && x.HasSqlAHUB && !x.HasWindowsAHUB)
                || (TypeFilter == "Both" && x.HasWindowsAHUB && x.HasSqlAHUB)
            )
        );

    private IEnumerable<HybridBenefitResource> OrderedResources =>
        SortColumn switch
        {
            "Health" => SortAscending ? FilteredResources.OrderBy(x => x.HealthStatus) : FilteredResources.OrderByDescending(x => x.HealthStatus),
            "Resource" => SortAscending ? FilteredResources.OrderBy(x => x.ResourceName) : FilteredResources.OrderByDescending(x => x.ResourceName),
            "Subscription" => SortAscending ? FilteredResources.OrderBy(x => x.Subscription) : FilteredResources.OrderByDescending(x => x.Subscription),
            "ResourceGroup" => SortAscending ? FilteredResources.OrderBy(x => x.ResourceGroup) : FilteredResources.OrderByDescending(x => x.ResourceGroup),
            "Windows" => SortAscending ? FilteredResources.OrderByDescending(x => x.HasWindowsAHUB) : FilteredResources.OrderBy(x => x.HasWindowsAHUB),
            "SQL" => SortAscending ? FilteredResources.OrderByDescending(x => x.HasSqlAHUB) : FilteredResources.OrderBy(x => x.HasSqlAHUB),
            "WindowsTag" => SortAscending ? FilteredResources.OrderByDescending(x => x.HasTagHB) : FilteredResources.OrderBy(x => x.HasTagHB),
            "SqlTag" => SortAscending ? FilteredResources.OrderByDescending(x => x.HasTagHBSQL) : FilteredResources.OrderBy(x => x.HasTagHBSQL),
            "Problem" => SortAscending ? FilteredResources.OrderBy(x => GetProblemText(x)) : FilteredResources.OrderByDescending(x => GetProblemText(x)),
            "ChangedBy" => SortAscending ? FilteredResources.OrderBy(x => x.LastChangedBy) : FilteredResources.OrderByDescending(x => x.LastChangedBy),
            "Sku" => SortAscending ? FilteredResources.OrderBy(x => x.VMSize) : FilteredResources.OrderByDescending(x => x.VMSize),
            "vCPUs" => SortAscending ? FilteredResources.OrderBy(x => x.vCPUs) : FilteredResources.OrderByDescending(x => x.vCPUs),
            "Activity" => SortAscending ? FilteredResources.OrderBy(x => x.ChangeType) : FilteredResources.OrderByDescending(x => x.ChangeType),
            "LastScan" => SortAscending ? FilteredResources.OrderBy(x => x.LastScanDate) : FilteredResources.OrderByDescending(x => x.LastScanDate),
            "Recurrence" => SortAscending ? FilteredResources.OrderBy(x => x.RecurrenceStatus) : FilteredResources.OrderByDescending(x => x.RecurrenceStatus),
            _ => FilteredResources.OrderBy(x => x.ResourceName)
        };

    private IEnumerable<HybridBenefitResource> PagedResources =>
        OrderedResources
            .Skip((CurrentPage - 1) * PageSize)
            .Take(PageSize);

    private int TotalPages =>
        Math.Max(1, (int)Math.Ceiling(FilteredResources.Count() / (double)PageSize));

    private void ClearFilters()
    {
        SearchText = "";
        SubscriptionSearch = "";
        StatusFilter = "";
        TypeFilter = "";
        SeverityFilter = "";
        CurrentPage = 1;
    }

    private void SortBy(string column)
    {
        if (SortColumn == column)
        {
            SortAscending = !SortAscending;
        }
        else
        {
            SortColumn = column;
            SortAscending = true;
        }

        CurrentPage = 1;
    }

    private string GetSortIcon(string column)
    {
        if (SortColumn != column)
        {
            return "";
        }

        return SortAscending ? "▲" : "▼";
    }

private async Task ExportExcelAsync()
    {
        var sb = new StringBuilder();
        sb.AppendLine("Health,Customer,Subscription,ResourceGroup,ResourceName,WindowsAHUB,SqlAHUB,TagHB,TagHBSQL,Problem,ChangeType,Severity,ChangeDate,LastScanDate");

        foreach (var item in FilteredResources)
        {
            sb.AppendLine(string.Join(",",
                Csv(item.HealthStatus),
                Csv(item.Customer),
                Csv(item.Subscription),
                Csv(item.ResourceGroup),
                Csv(item.ResourceName),
                Csv(item.HasWindowsAHUB ? "Yes" : "No"),
                Csv(item.HasSqlAHUB ? "Yes" : "No"),
                Csv(item.HasTagHB ? "OK" : "Missing"),
                Csv(item.HasTagHBSQL ? "OK" : "Missing"),
                Csv(GetProblemText(item)),
                Csv(item.ChangeType),
                Csv(item.Severity),
                Csv(item.ChangeDate?.ToString("dd/MM/yyyy HH:mm") ?? ""),
                Csv(item.LastScanDate?.ToString("dd/MM/yyyy HH:mm") ?? "")
            ));
        }

        var csv = sb.ToString();
        var fileName = $"Azure-Hybrid-Benefit-Resources-{DateTime.Now:yyyyMMddHHmm}.csv";

        await JS.InvokeVoidAsync("eval", $@"
            const blob = new Blob([{System.Text.Json.JsonSerializer.Serialize(csv)}], {{ type: 'text/csv;charset=utf-8;' }});
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = '{fileName}';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        ");
    }

    private static string Csv(string value)

    private async Task OpenHistoryAsync(HybridBenefitResource resource)
    {
        SelectedResource = resource;
        ShowHistoryPanel = true;
        IsLoadingHistory = true;
        ResourceHistory = new();

        try

    private static string GetRowClass(HybridBenefitResource item)

    private static string GetHealthClass(HybridBenefitResource item)
    {
        return item.HealthStatus switch

            "New" => "hb-health new",
            _ => "hb-health new"
        };

    {
        return item.HealthStatus switch
        {
            "Healthy" => "● Healthy",
            "Action Required" => "⚠ Action",
            "Changed" => "▲ Changed",
            "New" => "★ New",
            _ => "★ New"
        };
    }

    private static string GetProblemText(HybridBenefitResource item)
    {
        if (item.HasWindowsAHUB && !item.HasTagHB)
        {
            return "Falta TagHB";
        }

        if (item.HasSqlAHUB && !item.HasTagHBSQL)
        {
            return "Falta TagHBSQL";
        }

        if (item.ChangeType?.Contains("Removed", StringComparison.OrdinalIgnoreCase) == true)
        {
            return "Beneficio removido";
        }

        return "Sin problema";
    }

            return "hb-problem danger";

        return "hb-problem success";
