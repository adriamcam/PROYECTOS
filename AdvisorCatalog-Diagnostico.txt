using Azure.Security.KeyVault.Secrets;

var builder = WebApplication.CreateBuilder(args);

builder.Services
    .AddAuthentication(OpenIdConnectDefaults.AuthenticationScheme)
    .AddMicrosoftIdentityWebApp(builder.Configuration.GetSection("AzureAd"));

builder.Services.Configure<SqlSettings>(
    builder.Configuration.GetSection("SqlSettings"));

builder.Services.Configure<KeyVaultSettings>(
    builder.Configuration.GetSection("KeyVaultSettings"));

builder.Services.AddSingleton<SecretClient>(sp =>
{
    var keyVaultSettings = builder.Configuration
        .GetSection("KeyVaultSettings")
        .Get<KeyVaultSettings>();

    if (keyVaultSettings == null || string.IsNullOrWhiteSpace(keyVaultSettings.VaultName))
        throw new InvalidOperationException("KeyVaultSettings:VaultName no está configurado.");

    var vaultUri = new Uri($"https://{keyVaultSettings.VaultName}.vault.azure.net/");

    return new SecretClient(vaultUri, new DefaultAzureCredential());
});

builder.Services.AddAuthorization(options =>
{
    options.FallbackPolicy = null;
});

builder.Services
    .AddControllersWithViews(options =>
    {
        var policy = new AuthorizationPolicyBuilder()
            .RequireAuthenticatedUser()
            .Build();

        options.Filters.Add(new AuthorizeFilter(policy));
    })
    .AddMicrosoftIdentityUI();

builder.Services.AddScoped<IClientWorkspaceService, ClientWorkspaceService>();

builder.Services.AddRazorComponents()
    .AddInteractiveServerComponents();

builder.Services.AddItqsSocServices(builder.Configuration);

                                uniqueidentifier,

                        ORDER BY
                            PC.ExecutionDate DESC,
                            PC.Id DESC

                    ON TRY_CONVERT(

                    TenantId,

            cancellationToken: cancellationToken,
            commandTimeout: 120);

        var availableColumns =
            (await connection.QueryAsync<string>(
                columnsCommand))
            .ToHashSet(
                StringComparer.OrdinalIgnoreCase);

        static string? FindColumn(
            HashSet<string> columns,
            params string[] candidates)
        {
            return candidates.FirstOrDefault(
                columns.Contains);
        }

        static string QuoteColumn(string columnName)
        {
            return $"[{columnName.Replace("]", "]]")}]";
        }

        static string TextExpression(
            string? columnName,
            string fallback)
        {
            if (string.IsNullOrWhiteSpace(columnName))
            {
                return $"N'{fallback.Replace("'", "''")}'";
            }

            return
                $"COALESCE(NULLIF(CONVERT(nvarchar(4000), " +
                $"{QuoteColumn(columnName)}), N''), " +
                $"N'{fallback.Replace("'", "''")}')";
        }

        var titleColumn = FindColumn(
            availableColumns,
            "RecommendationName",
            "Recommendation",
            "ShortDescription",
            "RecommendationDescription",
            "Description",
            "Problem",
            "Title");

        var categoryColumn = FindColumn(
            availableColumns,
            "Category",
            "RecommendationCategory",
            "CategoryName");

        var impactColumn = FindColumn(
            availableColumns,
            "Impact",
            "ImpactLevel",
            "BusinessImpact",
            "RiskLevel");

        var resourceColumn = FindColumn(
            availableColumns,
            "ResourceName",
            "ImpactedResourceName",
            "ResourceDisplayName",
            "ImpactedResource",
            "ResourceId");

        var savingsColumn = FindColumn(
            availableColumns,
            "AnnualSavingsAmount");

        var updatedColumn = FindColumn(
            availableColumns,
            "AdvisorUpdatedAt",
            "LastUpdatedAt",
            "UpdatedAt",
            "LastUpdatedDateTime");

        if (savingsColumn is null)
        {
            throw new InvalidOperationException(
                "La vista dbo.vw_AdvisorRecommendationsValued " +
                "no contiene la columna AnnualSavingsAmount.");
        }

        var titleExpression = TextExpression(
            titleColumn,
            "Recomendación Advisor");

        var categoryExpression = TextExpression(
            categoryColumn,
            "Sin categoría");

        var impactExpression = TextExpression(
            impactColumn,
            "Sin clasificar");

        var resourceExpression = TextExpression(
            resourceColumn,
            "-");

        var updatedExpression =
            updatedColumn is null
                ? "CAST(NULL AS datetime2)"
                : $"TRY_CONVERT(datetime2, " +
                  $"{QuoteColumn(updatedColumn)})";

        var savingsExpression =
            $"ISNULL(TRY_CONVERT(decimal(19, 2), " +
            $"{QuoteColumn(savingsColumn)}), 0)";

        var sql = $"""
            SELECT TOP (@Top)
                Title =
                    {titleExpression},

                Category =
                    {categoryExpression},

                Impact =
                    {impactExpression},

                ResourceName =
                    {resourceExpression},

                AnnualSavingsAmount =
                    {savingsExpression},

                UpdatedAt =
                    {updatedExpression}

            FROM dbo.vw_AdvisorRecommendationsValued

            WHERE TenantId = @TenantId
              AND
              (
                  @SubscriptionId IS NULL
                  OR SubscriptionId = @SubscriptionId
              )

            ORDER BY
                {savingsExpression} DESC,
                {updatedExpression} DESC;
            """;

        var command = new CommandDefinition(
            sql,
            new
            {
                TenantId = tenantId,
                SubscriptionId = subscriptionId,
                Top = top
            },
            commandType: CommandType.Text,
            cancellationToken: cancellationToken,
            commandTimeout: 120);

        var recommendations =
            await connection.QueryAsync<ClientAdvisorRecommendation>(
                command);

        return recommendations.AsList();
    }

    public async Task<IReadOnlyList<ClientAdvisorScore>>
        GetAdvisorScoresAsync(
            Guid tenantId,
            Guid? subscriptionId,
            CancellationToken cancellationToken = default)
    {
        const string sql = """
            WITH LatestScores AS
            (
                SELECT
                    ScoreName,
                    Score,
                    ConsumptionUnits,
                    ImpactedResourceCount,
                    PotentialScoreIncrease,
                    LastScoreDateUtc,
                    LastSeenUtc,

                    RowNumber =
                        ROW_NUMBER() OVER
                        (
                            PARTITION BY
                                TRY_CONVERT(
                                    uniqueidentifier,
                                    SubscriptionId
                                ),
                                ScoreName

                            ORDER BY
                                LastScoreDateUtc DESC,
                                LastSeenUtc DESC
                        )

                FROM dbo.AzureAdvisorScores

                WHERE TRY_CONVERT(
                          uniqueidentifier,
                          TenantId
                      ) = @TenantId

                  AND
                  (
                      @SubscriptionId IS NULL
                      OR TRY_CONVERT(
                             uniqueidentifier,
                             SubscriptionId
                         ) = @SubscriptionId
                  )
            )

            SELECT
                ScoreName,

                Score =
                    CONVERT(
                        decimal(10, 2),
                        AVG(
                            CONVERT(
                                decimal(18, 4),
                                Score
                            )
                        )
                    ),

                ConsumptionUnits =
                    CONVERT(
                        decimal(18, 2),
                        SUM(
                            CONVERT(
                                decimal(18, 4),
                                ISNULL(ConsumptionUnits, 0)
                            )
                        )
                    ),

                ImpactedResourceCount =
                    SUM(ISNULL(ImpactedResourceCount, 0)),

                PotentialScoreIncrease =
                    CONVERT(
                        decimal(10, 2),
                        AVG(
                            CONVERT(
                                decimal(18, 4),
                                ISNULL(PotentialScoreIncrease, 0)
                            )
                        )
                    ),

                LastScoreDateUtc =
                    MAX(LastScoreDateUtc),

                LastSeenUtc =
                    MAX(LastSeenUtc)

            FROM LatestScores

            WHERE RowNumber = 1

            GROUP BY ScoreName

            ORDER BY
                CASE ScoreName
                    WHEN N'Cost' THEN 1
                    WHEN N'HighAvailability' THEN 2
                    WHEN N'OperationalExcellence' THEN 3
                    WHEN N'Performance' THEN 4
                    WHEN N'Security' THEN 5
                    ELSE 6
                END;
            """;

        using var connection =
            _connectionFactory.CreateConnection();

        var command = new CommandDefinition(
            sql,
            new
            {
                TenantId = tenantId,
                SubscriptionId = subscriptionId
            },
            commandType: CommandType.Text,
            cancellationToken: cancellationToken,
            commandTimeout: 120);

        var result =
            await connection.QueryAsync<ClientAdvisorScore>(
                command);

        return result.AsList();
    }

    public async Task<ClientSecurityScore?> GetSecurityScoreAsync(
        Guid tenantId,
        Guid? subscriptionId,
        CancellationToken cancellationToken = default)
    {
        const string sql = """
            WITH LatestSecurityScores AS
            (
                SELECT
                    SubscriptionId,
                    SecScore,
                    InsertedAt,

                    RowNumber =
                        ROW_NUMBER() OVER
                        (
                            PARTITION BY SubscriptionId
                            ORDER BY
                                InsertedAt DESC,
                                Id DESC
                        )

                FROM dbo.SecureScores

                WHERE TRY_CONVERT(
                          uniqueidentifier,
                          TenantId
                      ) = @TenantId

                  AND
                  (
                      @SubscriptionId IS NULL
                      OR TRY_CONVERT(
                             uniqueidentifier,
                             SubscriptionId
                         ) = @SubscriptionId
                  )
            )

            SELECT
                Score =
                    CONVERT(
                        decimal(10, 2),
                        ISNULL(
                            AVG(
                                CONVERT(
                                    decimal(18, 4),
                                    SecScore
                                )
                            ),
                            0
                        )
                    ),

                SubscriptionCount =
                    COUNT(*),

                UpdatedAt =
                    MAX(InsertedAt)

            FROM LatestSecurityScores

            WHERE RowNumber = 1;
            """;

        using var connection =
            _connectionFactory.CreateConnection();

        var command = new CommandDefinition(
            sql,
            new
            {
                TenantId = tenantId,
                SubscriptionId = subscriptionId
            },
            commandType: CommandType.Text,
            cancellationToken: cancellationToken,
            commandTimeout: 120);

        return await connection
            .QuerySingleOrDefaultAsync<ClientSecurityScore>(
                command);
    }

    private async Task ExportRecommendationsAsync()
    {
        var rows = FilteredRecommendations.ToList();

        if (rows.Count == 0)
        {
            return;
        }

        var csv = new StringBuilder();

        // BOM para que Excel reconozca correctamente tildes y caracteres UTF-8.
        csv.Append('\uFEFF');

        csv.AppendLine(
            "Suscripción;" +
            "Recomendación;" +
            "Descripción;" +
            "Categoría;" +
            "Impacto;" +
            "Recurso;" +
            "Resource ID;" +
            "Grupo de recursos;" +
            "Complejidad;" +
            "Ventana de mantenimiento;" +
            "Detalle de ventana;" +
            "Tiempo estimado;" +
            "Horas estimadas;" +
            "Minutos estimados;" +
            "Ahorro anual;" +
            "Moneda;" +
            "Fecha de actualización");

        foreach (var item in rows)
        {
            var values = new[]
            {
                item.SubscriptionName,
                item.RecommendationSpanish,
                item.DescriptionSpanish,
                TranslateCategory(item.AzureCategory),
                item.Impact,
                item.ResourceName,
                item.AffectedResource,
                item.ResourceGroup,
                item.Complexity,
                item.RequiresMaintenanceWindow ? "Sí" : "No",
                item.MaintenanceWindowText,
                FormatImplementationTime(item),
                item.ImplementationHours.ToString(
                    "0.##",
                    CultureInfo.InvariantCulture),
                item.ImplementationMinutes.ToString(
                    CultureInfo.InvariantCulture),
                item.AnnualSavingsAmount.ToString(
                    "0.00",
                    CultureInfo.InvariantCulture),
                item.Currency,
                item.AdvisorUpdatedAt?.ToString(
                    "yyyy-MM-dd HH:mm:ss",
                    CultureInfo.InvariantCulture)
                    ?? string.Empty
            };

            csv.AppendLine(
                string.Join(
                    ";",
                    values.Select(EscapeCsvValue)));
        }

        var bytes = Encoding.UTF8.GetBytes(csv.ToString());

        await using var stream = new MemoryStream(bytes);
        using var streamReference = new DotNetStreamReference(stream);

        var timestamp = DateTime.Now.ToString(
            "yyyyMMdd-HHmmss",
            CultureInfo.InvariantCulture);

        var fileName =
            $"Azure-Advisor-Recomendaciones-{timestamp}.csv";

        await JS.InvokeVoidAsync(
            "downloadFileFromStream",
            fileName,
            streamReference);
    }

    private static string EscapeCsvValue(object? value)
    {
        var text = Convert.ToString(
                value,
                CultureInfo.InvariantCulture)
            ?? string.Empty;

        text = text
            .Replace("\r\n", " ")
            .Replace("\n", " ")
            .Replace("\r", " ")
            .Replace("\"", "\"\"");

        return $"\"{text}\"";
    }

    private IReadOnlyList<ClientAdvisorRecommendation>
        FilteredRecommendations
    {
        get
        {
            IEnumerable<ClientAdvisorRecommendation> query =
                SummaryFilter switch
                {
                    "resources" =>
                        Recommendations
                            .Where(
                                x =>
                                    !string.IsNullOrWhiteSpace(
                                        x.AffectedResource))
                            .GroupBy(
                                x => x.AffectedResource,
                                StringComparer.OrdinalIgnoreCase)
                            .Select(
                                group =>
                                    group
                                        .OrderByDescending(
                                            x =>
                                                x.AnnualSavingsAmount)
                                        .ThenBy(
                                            x =>
                                                x.RecommendationSpanish)
                                        .First()),

                    "valued" =>
                        Recommendations.Where(
                            x => x.AnnualSavingsAmount > 0),

                    _ =>
                        Recommendations
                };

            if (!string.IsNullOrWhiteSpace(SearchText))
            {
                var search = SearchText.Trim();

                query = query.Where(
                    x =>
                        Contains(x.RecommendationSpanish, search) ||
                        Contains(x.DescriptionSpanish, search) ||
                        Contains(x.ResourceName, search) ||
                        Contains(x.ResourceGroup, search) ||
                        Contains(x.SubscriptionName, search));
            }

            if (!string.IsNullOrWhiteSpace(SelectedCategory))
            {
                query = query.Where(
                    x => string.Equals(
                        x.AzureCategory,
                        SelectedCategory,
                        StringComparison.OrdinalIgnoreCase));
            }

            if (!string.IsNullOrWhiteSpace(SelectedImpact))
            {
                query = query.Where(
                    x => string.Equals(
                        x.Impact,
                        SelectedImpact,
                        StringComparison.OrdinalIgnoreCase));
            }

            if (OnlyWithSavings)
            {
                query = query.Where(
                    x => x.AnnualSavingsAmount > 0);
            }

            return query.ToArray();
        }

    protected override async Task OnParametersSetAsync()
    {
        if (Context?.TenantId is not Guid tenantId)

            return;

    private async Task LoadDataAsync()
    {
        if (Context?.TenantId is not Guid tenantId)
        {
            ClearData();
            return;
        }

        IsLoading = true;
        ErrorMessage = null;

        try
        {
            var recommendationsTask =
                WorkspaceService.GetAdvisorRecommendationsAsync(
                    tenantId,
                    Context.SubscriptionId);

            var advisorScoresTask =
                WorkspaceService.GetAdvisorScoresAsync(
                    tenantId,
                    Context.SubscriptionId);

            var securityScoreTask =
                WorkspaceService.GetSecurityScoreAsync(
                    tenantId,
                    Context.SubscriptionId);

            await Task.WhenAll(
                recommendationsTask,
                advisorScoresTask,
                securityScoreTask);

            Recommendations = await recommendationsTask;
            AdvisorScores = await advisorScoresTask;
            SecurityScore = await securityScoreTask;

            LoadedTenantId = tenantId;
            LoadedSubscriptionId = Context.SubscriptionId;

            SummaryFilter = "all";
            SortColumn = string.Empty;
            SortAscending = true;
            CurrentPage = 1;
        }
        catch (Exception ex)
        {
            Recommendations =
                Array.Empty<ClientAdvisorRecommendation>();

            AdvisorScores =
                Array.Empty<ClientAdvisorScore>();

            SecurityScore = null;

        SelectedRecommendation = null;

            ErrorMessage = ex.Message;
        }
        finally
        {
            IsLoading = false;
        }
    }

                break;

                SortColumn = string.Empty;
                SortAscending = true;

    private IReadOnlyList<ClientWorkspaceCustomer> Customers =
        Array.Empty<ClientWorkspaceCustomer>();

    private IReadOnlyList<ClientWorkspaceSubscription> Subscriptions =
        Array.Empty<ClientWorkspaceSubscription>();

    private ClientWorkspaceContext WorkspaceContext { get; } = new();

    private ClientWorkspaceOverview Overview { get; set; } = new();

    private string SelectedCustomerId { get; set; } = string.Empty;

    private string CustomerInputText { get; set; } = string.Empty;

    private string SearchText { get; set; } = string.Empty;

    private string SelectedSubscriptionId { get; set; } = string.Empty;

    private string SelectedTab { get; set; } = "overview";

    private bool IsLoading { get; set; }

    private string ErrorMessage { get; set; } = string.Empty;

    private IEnumerable<ClientWorkspaceCustomer> FilteredCustomers
    {
        get
        {
            if (string.IsNullOrWhiteSpace(SearchText))
            {
                return Customers
                    .OrderBy(customer => customer.DisplayName);
            }

            var search =
                SearchText.Trim();

            return Customers
                .Where(customer =>
                    ContainsIgnoreCase(
                        customer.CustomerName,
                        search
                    )
                    ||
                    ContainsIgnoreCase(
                        customer.CustomerNamePortal,
                        search
                    )
                    ||
                    ContainsIgnoreCase(
                        customer.SubscriptionName,
                        search
                    )
                    ||
                    ContainsIgnoreCase(
                        customer.TenantId.ToString(),
                        search
                    )
                    ||
                    ContainsIgnoreCase(
                        customer.PartnerTenant,
                        search
                    )
                    ||
                    ContainsIgnoreCase(
                        customer.GdapStatus,
                        search
                    )
                )
                .OrderBy(customer => customer.DisplayName);
        }
    }

    private string DisplayCustomerName =>
        WorkspaceContext.HasCustomer
            ? (
                string.IsNullOrWhiteSpace(
                    WorkspaceContext.CustomerName
                )
                    ? "Cliente sin nombre"
                    : WorkspaceContext.CustomerName
              )
            : "No seleccionado";

    private string DisplayTenantId =>
        WorkspaceContext.TenantId?.ToString()
        ?? "-";

    private string DisplayPartnerTenant =>
        string.IsNullOrWhiteSpace(
            WorkspaceContext.PartnerTenant
        )
            ? "-"
            : WorkspaceContext.PartnerTenant;

    private string DisplaySubscriptionName =>
        WorkspaceContext.HasSubscription
            ? WorkspaceContext.SubscriptionName
            : (
                WorkspaceContext.HasCustomer
                    ? "Todas las suscripciones"
                    : "No seleccionada"
              );

    private string DisplaySubscriptionState =>
        WorkspaceContext.HasSubscription
            ? (
                string.IsNullOrWhiteSpace(
                    WorkspaceContext.SubscriptionState
                )
                    ? "Disponible"
                    : WorkspaceContext.SubscriptionState
              )
            : (
                WorkspaceContext.HasCustomer
                    ? Overview.OverallStatus
                    : "-"
              );

    protected override async Task OnInitializedAsync()
    {
        await LoadCustomersAsync();
    }

    private async Task LoadCustomersAsync()
    {
        IsLoading = true;
        ErrorMessage = string.Empty;

        try
        {
            Customers =
                await WorkspaceService.GetCustomersAsync();
        }
        catch (Exception ex)
        {
            ErrorMessage =
                $"No fue posible cargar los clientes: {ex.Message}";
        }
        finally
        {
            IsLoading = false;
        }
    }

    private async Task OnCustomerInputChangedAsync(
        ChangeEventArgs args)
    {
        CustomerInputText =
            Convert.ToString(args.Value)?.Trim()
            ?? string.Empty;

        if (string.IsNullOrWhiteSpace(CustomerInputText))
        {
            ClearCustomerSelection();
            return;
        }

        var customer =
            Customers.FirstOrDefault(item =>
                string.Equals(
                    item.DisplayName,
                    CustomerInputText,
                    StringComparison.OrdinalIgnoreCase
                )
                ||
                string.Equals(
                    item.TenantId.ToString(),
                    CustomerInputText,
                    StringComparison.OrdinalIgnoreCase
                )
            );

        if (customer is null)
        {
            var matches =
                FilteredCustomers
                    .Where(item =>
                        ContainsIgnoreCase(
                            item.DisplayName,
                            CustomerInputText
                        )
                        ||
                        ContainsIgnoreCase(
                            item.SubscriptionName,
                            CustomerInputText
                        )
                    )
                    .Take(2)
                    .ToList();

            if (matches.Count == 1)
            {
                customer = matches[0];
            }
        }

        if (customer is null)
        {
            ErrorMessage =
                "Seleccione un cliente de la lista de resultados.";

            return;
        }

        await SelectCustomerAsync(customer);
    }

    private async Task SelectCustomerAsync(
        ClientWorkspaceCustomer customer)
    {
        ErrorMessage = string.Empty;

        SelectedCustomerId =
            customer.TenantId.ToString();

        CustomerInputText =
            customer.DisplayName;

        SelectedSubscriptionId = string.Empty;

        Subscriptions =
            Array.Empty<ClientWorkspaceSubscription>();

        WorkspaceContext.Clear();
        Overview = new ClientWorkspaceOverview();

        WorkspaceContext.TenantId =
            customer.TenantId;

        WorkspaceContext.CustomerName =
            customer.CustomerName;

        WorkspaceContext.CustomerNamePortal =
            customer.CustomerNamePortal;

        WorkspaceContext.PartnerTenant =
            customer.PartnerTenant;

        WorkspaceContext.GdapStatus =
            customer.GdapStatus;

        await LoadCustomerDataAsync();
    }

    private void ClearCustomerSelection()
    {
        SelectedCustomerId = string.Empty;
        CustomerInputText = string.Empty;
        SelectedSubscriptionId = string.Empty;

        Subscriptions =
            Array.Empty<ClientWorkspaceSubscription>();

        WorkspaceContext.Clear();
        Overview = new ClientWorkspaceOverview();
        ErrorMessage = string.Empty;
    }

    private void OnSearchChanged(ChangeEventArgs args)
    {
        SearchText =
            Convert.ToString(args.Value)
            ?? string.Empty;
    }

    private async Task OnSubscriptionChangedAsync(
        ChangeEventArgs args)
    {
        SelectedSubscriptionId =
            Convert.ToString(args.Value) ?? string.Empty;

        WorkspaceContext.SubscriptionId = null;
        WorkspaceContext.SubscriptionName = string.Empty;
        WorkspaceContext.SubscriptionState = string.Empty;

        if (Guid.TryParse(
                SelectedSubscriptionId,
                out var subscriptionId))
        {
            var subscription =
                Subscriptions.FirstOrDefault(
                    item =>
                        item.SubscriptionId == subscriptionId);

            if (subscription is not null)
            {
                WorkspaceContext.SubscriptionId =
                    subscription.SubscriptionId;

                WorkspaceContext.SubscriptionName =
                    subscription.SubscriptionName;

                WorkspaceContext.SubscriptionState =
                    subscription.State;
            }
        }

        await LoadOverviewAsync();
    }

    private async Task LoadCustomerDataAsync()
    {
        if (!WorkspaceContext.TenantId.HasValue)
        {
            return;
        }

        IsLoading = true;
        ErrorMessage = string.Empty;

        try
        {
            Subscriptions =
                await WorkspaceService.GetSubscriptionsAsync(
                    WorkspaceContext.TenantId.Value);

            await LoadOverviewInternalAsync();
        }
        catch (Exception ex)
        {
            ErrorMessage =
                $"No fue posible cargar el Workspace: {ex.Message}";
        }
        finally
        {
            IsLoading = false;
        }
    }

    private async Task LoadOverviewAsync()
    {
        if (!WorkspaceContext.TenantId.HasValue)
        {
            return;
        }

        IsLoading = true;
        ErrorMessage = string.Empty;

        try
        {
            await LoadOverviewInternalAsync();
        }
        catch (Exception ex)
        {
            ErrorMessage =
                $"No fue posible cargar el resumen: {ex.Message}";
        }
        finally
        {
            IsLoading = false;
        }
    }

    private async Task LoadOverviewInternalAsync()
    {
        if (!WorkspaceContext.TenantId.HasValue)
        {
            return;
        }

        Overview =
            await WorkspaceService.GetOverviewAsync(
                WorkspaceContext.TenantId.Value,
                WorkspaceContext.SubscriptionId);
    }

    private async Task RefreshAsync()
    {
        if (!WorkspaceContext.TenantId.HasValue)
        {
            await LoadCustomersAsync();
            return;
        }

        await LoadCustomerDataAsync();
    }

    private static bool ContainsIgnoreCase(
        string? source,
        string search)
    {
        return !string.IsNullOrWhiteSpace(source)
            && source.Contains(
                search,
                StringComparison.OrdinalIgnoreCase
            );
    }

    private static string GetCustomerOptionDescription(
        ClientWorkspaceCustomer customer)
    {
        var subscription =
            string.IsNullOrWhiteSpace(customer.SubscriptionName)
                ? "Sin suscripción"
                : customer.SubscriptionName;

        var partner =
            string.IsNullOrWhiteSpace(customer.PartnerTenant)
                ? "Sin partner"
                : customer.PartnerTenant;

        return $"{subscription} | {partner}";
    }

    private void SelectTab(string tab)
    {
        SelectedTab = tab;
    }

    private string GetTabClass(string tab)
    {
        return SelectedTab == tab
            ? "workspace-tab active"
            : "workspace-tab";
    }
}
